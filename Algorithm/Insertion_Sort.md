    numbers.append(int(input()))

# Insertion Sort
for i in range(1, n):
    for j in range(i, 0, -1):      
        if numbers[j - 1] > numbers[j]:       # 1. 
            numbers[j], numbers[j - 1] = numbers[j - 1], numbers[j]     # 2.

    print(n)
